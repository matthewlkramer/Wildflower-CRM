Validate Claude’s proposed actions locally before saving them.
Add organization candidates, not just funder candidates.
Make auto-suppression safer.

The file currently asks Claude for structured CRM mutations and stores them directly as proposedActions; the comment says the downstream dispatcher re-validates IDs, but this module can still save invalid or low-confidence actions into the review queue. That preserves the error but allows the row to be picked up again.

Second important issue: suppression update can leave sentinel rows stuck

This block:

.where(
  and(
    eq(emailProposals.id, proposalId),
    shouldIgnore ? eq(emailProposals.status, "pending") : sql`true`,
  ),
);

If shouldIgnore === true but the proposal status is no longer pending, the update affects zero rows. That means the row may remain stuck with the epoch sentinel.

Use two updates instead: one always records the analysis; the second conditionally auto-ignores only pending rows.

    actionsError: null,
    updatedAt: new Date(),
  })
  .where(eq(emailProposals.id, proposalId));

  await db
    .update(emailProposals)

        suppress?.reason ?? "non-actionable noise"
      }`.slice(0, 500),
      updatedAt: new Date(),
    })
    .where(
      and(
        eq(emailProposals.id, proposalId),

Add local validation before saving actions

Claude’s tool schema ensures shape, but not CRM correctness. I would add a sanitizeProposedActions() step after parsing the tool response and before saving.

Add this helper
function normalizeEmail(value: string): string {
  return value.trim().toLowerCase();
}

function exactlyOne(values: unknown[]): boolean {
  return values.filter((v) => typeof v === "string" && v.trim().length > 0).length === 1;
}

function truncateReason(reason: string | undefined): string {
  return (reason || "AI-proposed CRM update").slice(0, 140);
}

function sanitizeProposedActions(args: {
  rawActions: ProposedAction[];
  personContext: PersonContext | null;
  funderTargetId: string | null;
  funderCandidates: FunderCandidate[];
  organizationCandidates?: { id: string; name: string }[];
}): ProposedAction[] {
  const {
    rawActions,
    personContext,
    funderTargetId,
    funderCandidates,
    organizationCandidates = [],
  } = args;

  const allowedPersonIds = new Set<string>();
  const allowedPerIds = new Set<string>();
  const allowedEmailIds = new Set<string>();
  const existingEmails = new Set<string>();
  const allowedFunderIds = new Set<string>();
  const allowedOrganizationIds = new Set<string>();

  if (personContext) {
    allowedPersonIds.add(personContext.id);

    for (const e of personContext.emails) {
      allowedEmailIds.add(e.id);
      existingEmails.add(normalizeEmail(e.email));
    }

    for (const r of personContext.roles) {
      allowedPerIds.add(r.id);
      if (r.funderId) allowedFunderIds.add(r.funderId);
      if (r.organizationId) allowedOrganizationIds.add(r.organizationId);
    }
  }

  if (funderTargetId) allowedFunderIds.add(funderTargetId);
  for (const f of funderCandidates) allowedFunderIds.add(f.id);
  for (const o of organizationCandidates) allowedOrganizationIds.add(o.id);

  const clean: ProposedAction[] = [];

  for (const action of rawActions) {
    const reason = truncateReason(action.reason);

    switch (action.type) {
      case "deactivate_per": {
        if (!allowedPerIds.has(action.perId)) break;
        clean.push({ ...action, reason });
        break;
      }

      case "create_per": {
        if (!allowedPersonIds.has(action.personId)) break;

        const hasExactlyOneEntity = exactlyOne([
          action.funderId,
          action.organizationId,
          action.paymentIntermediaryId,
          action.householdId,
        ]);

        if (!hasExactlyOneEntity) break;
        if (action.funderId && !allowedFunderIds.has(action.funderId)) break;
        if (action.organizationId && !allowedOrganizationIds.has(action.organizationId)) break;

        clean.push({ ...action, reason });
        break;
      }

      case "create_person_with_per": {
        const hasExactlyOneEntity = exactlyOne([
          action.funderId,
          action.organizationId,
        ]);

        if (!hasExactlyOneEntity) break;
        if (action.funderId && !allowedFunderIds.has(action.funderId)) break;
        if (action.organizationId && !allowedOrganizationIds.has(action.organizationId)) break;

        if (!action.firstName.trim() || !action.lastName.trim()) break;

        clean.push({
          ...action,
          emailAddress: action.emailAddress
            ? normalizeEmail(action.emailAddress)
            : undefined,
          reason,
        });
        break;
      }

      case "add_email": {
        if (!allowedPersonIds.has(action.personId)) break;

        const emailAddress = normalizeEmail(action.emailAddress);
        if (!emailAddress.includes("@")) break;
        if (existingEmails.has(emailAddress)) break;

        clean.push({
          ...action,
          emailAddress,
          reason,
        });
        break;
      }

      case "set_primary_email": {
        const hasExistingEmailId =
          action.emailId && allowedEmailIds.has(action.emailId);

        const hasPersonEmailTuple =
          action.personId &&
          action.emailAddress &&
          allowedPersonIds.has(action.personId) &&
          existingEmails.has(normalizeEmail(action.emailAddress));

        if (!hasExistingEmailId && !hasPersonEmailTuple) break;

        clean.push({
          ...action,
          emailAddress: action.emailAddress
            ? normalizeEmail(action.emailAddress)
            : undefined,
          reason,
        });
        break;
      }

      case "mark_email_invalid": {
        const emailAddress = normalizeEmail(action.emailAddress);

        // Prevent the model from invalidating arbitrary emails not already
        // associated with the matched person.
        if (personContext && !existingEmails.has(emailAddress)) break;

        clean.push({
          ...action,
          emailAddress,
          reason,
        });
        break;
      }

      case "create_grant_opportunity": {
        if (action.funderId && !allowedFunderIds.has(action.funderId)) break;

        clean.push({
          ...action,
          title: action.title.trim().slice(0, 250),
          reason,
        });
        break;
      }
    }
  }

  return clean;
}

async function findOrganizationCandidates(
  name: string | null | undefined,
): Promise<OrganizationCandidate[]> {
  if (!name || name.trim().length < 3) return [];

  const term = name.trim();

  const rows = await db
    .select({ id: organizations.id, name: organizations.name })
    .from(organizations)
    .where(
      or(
        ilike(organizations.name, term),
        ilike(organizations.name, `%${term}%`),
      ),
    )
    .limit(5);

  return rows
    .filter((r): r is { id: string; name: string } => r.name !== null)
    .map((r) => ({ id: r.id, name: r.name }));
}

Tighten suppression

Auto-suppression is high-impact because it hides items from the reviewer. I would add a local guard so that suppression only applies for proposal kinds where suppression is safe.

const SUPPRESSIBLE_KINDS = new Set([
  "grant_opportunity",
  "auto_responder",
  "signature_update",
  "bounce",
]);

function canAutoSuppress(proposal: EmailProposal, actions: ProposedAction[]): boolean {
  if (!SUPPRESSIBLE_KINDS.has(String(proposal.kind))) return false;

  // If the model found a concrete CRM mutation, do not hide the proposal.
  if (actions.length > 0) return false;

  return true;
}

Then change:

const shouldIgnore = suppress?.shouldSuppress === true;

to:

const shouldIgnore =

This prevents a model response from both proposing a meaningful CRM action and hiding the item.
